\begin{frame}{扩展主题宏包}
  \textbf{将自定义配置封装于扩展宏包 \pkg{extension-xxx.sty}, 而后传递给 SCU Beamer Theme.} 例如新三国天意爷主题, 宏包命名可为 \pkg{extension-TianYiYe.sty}.\\[.5ex]

  \structure{核心步骤}
  \begin{enumerate}
    \item 在项目根目录下创建 \pkg{extension-xxx.sty}
    \item 在宏包中定义颜色 \textsc{Color}、字体 \textsc{Font}、背景 \textsc{Background}、视觉识别 \textsc{VI}
    \item 通过 \cmd{PassOptionsToPackage} 将选项传递至 SCU Beamer Theme (即主题主宏包 \pkg{beamerthemescu})
    \item 在 \texttt{.tex} 文件导言区调用扩展宏包, 须在引入 SCU Beamer Theme 之前 (即使用 \cmd{usetheme} 之前)
  \end{enumerate}\vspace*{1ex}

  \structure{自定义相关的 SCU 主题选项}\vspace*{.5ex}

      \texttt{VI} & \texttt{Custom} & 激活视觉识别自定义模式 \\
      \texttt{MathFont} & \marg{font} & 自定义数学字体 \\
      \texttt{LogoWidth} & \marg{length} & 调整页脚 Logo 背景宽度(默认\texttt{48pt}) \\
    \end{tabular*}
  \end{table}\vspace*{-1ex}
\end{frame}

\begin{frame}{自定义字体 \scugoback{back:CustomFontTheme}{BACK}}\label{goto:CustomFontTheme}
  \structure{文本字体设置}\\[.5ex]
  \textbf{在扩展宏包中定义 \cmd{beamer@scu@fontextend} 宏, 需传递选项 \texttt{FontTheme=Custom} 至 \pkg{beamerthemescu}.} 主题通过该宏加载自定义的中英文无衬线字体(\texttt{sans})、等宽字体(\texttt{mono}), 并可为中文配置 \texttt{Italic} 字形(通常用楷体).\medskip

  \begin{columns}[T, onlytextwidth]
    \begin{column}{.64\textwidth}
      如下示例将西文无衬线字体设为 Cinzel, 等宽字体设为 Latin Modern Mono (斜体用 Slanted 变体), 中文无衬线字体设为方正姚体, 中文斜体和等宽字体均设为方正舒体. 进一步自定义字体, 可参考 \pkg{beamerfontthemescu.sty}.\\[.5ex]
      \cmd{def}\cmd{beamer@scu@fontextend}\marg*{\%\\
      \hspace*{1em}\cmd{setsansfont}\marg*{Cinzel}\%\hfill\textsf{\scshape Sans}\\
      \hspace*{1em}\cmd{setmonofont}\marg*{Latin Modern Mono}\%\hfill\textsf{\scshape Mono}\\
      \hspace*{2em}\oarg*{ItalicFont = \marg*{Latin Modern Mono Slanted}}\%\hfill\textsf{\scshape Mono (Italic)}\\
      \hspace*{1em}\cmd{setCJKsansfont}\marg*{FZYaoTi}\%\hfill\textsf{\scshape CJK Sans}\\
      \hspace*{2em}\oarg*{ItalicFont = \marg*{FZShuTi}}\%\hfill\textsf{\scshape CJK Sans (Italic)}\\
      \hspace*{1em}\cmd{setCJKmonofont}\marg*{FZShuTi}\%\hfill\textsf{\scshape CJK Mono}\\
      }\bigskip
    \end{column}\qquad
    \begin{column}{.32\textwidth}
      如需查询本机已安装的字体名称:\\[1ex]
      \textbf{Windows}\\
      \textsc{Win + R} 打开运行, 输入 \texttt{ms-settings:fonts}\\[.5ex]
      \textbf{macOS}\\
      系统自带字体本(Font Book)应用\\[.5ex]
      \textbf{Linux}\\
      可使用 Fontconfig 的 \texttt{fc-list} 查询\\
      \texttt{fc-list}\\
      \texttt{fc-list :lang=zh}\\
      \texttt{fc-list | grep "FontName"}
    \end{column}
  \end{columns}
  
  \structure{数学字体设置}\\[.5ex]
  \textbf{数学字体可通过 \cmd{usetheme} 选项直接配置, 可取 \texttt{XITS}、\texttt{LM} 或任意自定义字体名称.} 通过以下命令传递至 SCU Beamer Theme (调用位置参见 \hyperref[frame:PassOpts]{\textbf{传递选项与调用宏包}}).\\
  \cmd{PassOptionsToPackage}\marg*{MathFont=\marg{font}}\marg*{beamerthemescu}
\end{frame}

\begin{frame}{自定义背景及视觉识别标识 \scugoback{back:CustomBackgroundTheme}{BACK}}\label{goto:CustomBackgroundTheme}
  \begin{columns}[T, onlytextwidth]
    \begin{column}{.32\textwidth}
      \structure{背景图片}\\[.5ex]
      \textbf{在扩展宏包中定义以下宏, 设置各封面、目录和正文的背景图片路径, 需传递选项 \texttt{Background=Custom} 至 \pkg{beamerthemescu}.} 主题将以 \texttt{width=\cmd{paperwidth}} 加载背景图.\\[1ex]

      \texttt{\% bg:: Frame 背景.}\\
      \cmd{def}\cmd{scu@path@bg}\marg*{\%\\
      \hspace*{1em}./resources/frame.png}\\[.5ex]
      \texttt{\% bgoftitle:: 封面背景.}\\
      \cmd{def}\cmd{scu@path@bgoftitle}\marg*{\%\\
      \hspace*{1em}./resources/title.png}\\[.5ex]
      \texttt{\% bgofsubsectoc:: 小节目录背景.}\\
      \cmd{def}\cmd{scu@path@bgofsubsectoc}\marg*{\%\\
      \hspace*{1em}./resources/ouline.png}
    \end{column}\hfil
    \begin{column}{.64\textwidth}
      \structure{视觉识别标识}\\[.5ex]
      \textbf{在扩展宏包中定义 \cmd{beamer@scu@viextend} 宏, 设置如下4种视觉识别标识, 需传递选项 \texttt{VI=Custom} 至 \pkg{beamerthemescu}.} 各标识均建议手动调整宽/高. 此外, 页脚 Logo 背景宽度可通过 \texttt{LogoWidth=100pt} 传递至 \pkg{beamerthemescu}(默认\texttt{48pt}, 调用位置参见 \hyperref[frame:PassOpts]{\textbf{传递选项与调用宏包}}).\\[1ex]

      \cmd{def}\cmd{beamer@scu@viextend}\marg*{\%\\
      \hspace*{1em}\texttt{\% Logo:: name.}\\
      \hspace*{1em}\cmd{pgfdeclareimage}\oarg*{width=40pt}\marg*{lg}\marg*{./resources/name.pdf}\%\\
      \hspace*{1em}\texttt{\% Logo+Name:: logo+name.}\\
      \hspace*{1em}\cmd{pgfdeclareimage}\oarg*{height=36pt}\marg*{lg-name}\marg*{./resources/logo+name.pdf}\%\\
      \hspace*{1em}\texttt{\% Landmark:: building.}\\
      \hspace*{1em}\cmd{pgfdeclareimage}\oarg*{height=80pt}\marg*{landmark}\marg*{./resources/building.png}\%\\
      \hspace*{1em}\texttt{\% Verify:: verify.}\\
      \hspace*{1em}\cmd{pgfdeclareimage}\oarg*{height=30pt}\marg*{verify}\marg*{./resources/verify.png}\%\\
      }
    \end{column}
  \end{columns}\medskip
  \textit{注: 文件名避免使用下划线, draft 模式下会被识别为数学模式字符, 导致报错.}
\end{frame}

\begin{frame}{传递选项与调用宏包}\label{frame:PassOpts}
  \begin{columns}[T, onlytextwidth]
    \begin{column}{.5\textwidth}
      \structure{传递选项}\\[.5ex]
      在扩展宏包末尾传递自定义选项至 SCU 主题.\\[1ex]

      \cmd{PassOptionsToPackage}\cmd{\%}\\
      \hspace*{1em}\texttt{LogoWidth=100pt,} \texttt{\% 页脚 Logo 背景宽度}\\
      \hspace*{1em}\texttt{FontTheme=Custom,} \texttt{\% 字体自定义模式}\\
      \hspace*{1em}\texttt{MathFont=\{Fira Math\},} \texttt{\% 数学字体}\\
      \hspace*{1em}\texttt{ColorDisplay=Custom,} \texttt{\% 颜色自定义模式}\\
      \hspace*{1em}\texttt{Background=Custom,} \texttt{\% 背景自定义模式}\\
      \hspace*{1em}\texttt{VI=Custom} \texttt{\% 视觉识别自定义模式}\\
      \marg*{beamerthemescu}
    \end{column}
    \begin{column}{.5\textwidth}
      \structure{调用宏包}\\[.5ex]
      在 \texttt{.tex} 导言区调用自定义扩展宏包.\\[1ex]

      \cmd{PassOptionsToPackage}\marg*{cmyk}\marg*{xcolor}\texttt{\% 解决设置 CMYK 颜色时生成 PDF 的色彩偏移}\\
      \cmd{documentclass}\oarg*{\%
      \hspace*{1em}\% draft,\% 草稿模式\\
      \hspace*{1em}\% handout,\% 讲义模式\\
      \hspace*{1em}\% aspectratio=169,\% 演示比例(推荐) 16:9\\
      \hspace*{1em}hyperref, UTF8, CJK\%\\
      }\marg*{beamer}\\[.5ex]
      \texttt{\% --------SCU Beamer 模板宏包--------}\\
      \cmd{usepackage}\marg*{extension-TianYiYe}\\
      \cmd{usetheme}\oarg*{...}\marg*{scu}\\[1ex]
    \end{column}\bigskip
  \end{columns}
  扩展宏包须在 \cmd{usetheme} 之前调用, 且不要在 \cmd{usetheme} 选项中覆盖扩展宏包已传递的参数.
\end{frame}
